/*1 - Faca um programa que carregue um vetor com 10 numeros reais.
Calcule e mostre a quantidade de numeros negativos e a soma dos numeros positivos deste vetor.*/

import java.util.Scanner;

public class CalculoNegativosESomaPositivos {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        
        // Crie um vetor de numeros reais com 10 posicoes
        double[] vetor = new double[10];
        
        // Carregue o vetor com valores fornecidos pelo usuario
        for (int i = 0; i < vetor.length; i++) {
            System.out.print("Digite um numero real: ");
            vetor[i] = scanner.nextDouble();
        }
        
        // Inicialize variaveis para contar numeros negativos e somar numeros positivos
        int quantidadeNegativos = 0;
        double somaPositivos = 0.0;
        
        // Percorra o vetor para calcular a quantidade de negativos e a soma dos positivos
        for (int i = 0; i < vetor.length; i++) {
            if (vetor[i] < 0) {
                quantidadeNegativos++;
            } else if (vetor[i] > 0) {
                somaPositivos += vetor[i];
            }
        }
        
        // Exiba os resultados
        System.out.println("Quantidade de numeros negativos: " + quantidadeNegativos);
        System.out.println("Soma dos numeros positivos: " + somaPositivos);
        
        scanner.close();
    }
}
